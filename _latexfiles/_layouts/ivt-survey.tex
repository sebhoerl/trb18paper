\providecommand{\createtitlepage}{
    \pagenumbering{arabic}%
    \setcounter{page}{0}%
    \providecommand{\mysubtitle}{}%
    \begin{titlepage}
        \setlength{\parfillskip}{0cm plus 30cm}
        \mytitlepagefontcommand
        \hrule
        \noindent\parbox[][\mytitlefigureboxheight][c]{\textwidth}{
            \ifdefined\mytitletext
            \begin{center}\mytitletext\end{center}%
            \else
            \ifthenelse
            {\equal{\mytitlefigure}{}}
            {}
            {\begin{center}\includegraphics[width=\textwidth,totalheight=9cm,keepaspectratio=true]{\mytitlefigure}\end{center}}
            \fi
        }
        \vspace{0.1in}
      
        \hrule
        
        \vspace{0.2in}
        
        \begin{singlespace}
            \noindent\LARGE\textbf{\mytitle}
            
            \ifthenelse{\equal{\mysubtitle}{}}
            {}
            {\Large\textbf{\mysubtitle}}
        \end{singlespace}
        
        \vspace{0.3in}
        
        %    \noindent\parbox[][3cm][l]{\textwidth}{
        %      \large\textbf{\internauthorlist}
        %    }
        
        
        \vfill
        %\vspace{2.5in}
        
        \noindent
        \begin{singlespace}
            \begin{tabular*}{\textwidth}{@{}l@{\extracolsep{\fill}}r@{}}
                \myinternpapertypeandnumber
                \textbf\interninstitution & \textbf\mytitlepagedate \\
            \end{tabular*}
        \end{singlespace}
        
        \vspace{0.25in}
        
        \mylogos
    \end{titlepage}
}

\providecommand{\myheaderfooterdef}{
    \usepackage{scrpage2}%
    \clearscrheadfoot%
    \automark[section]{chapter}%
    %\renewcommand{\headrulewidth}{}%
    %\ohead{\slshape \footnotesize \squelch{\wordmonth~\myyear}}%
    %\ihead{\providecommand{\myheadinglogos}{\myheadinglogos}\slshape \footnotesize \squelch{\myheadinglogos}}%
    \ihead[\providecommand{\myheadinglogos}{\myheadinglogos}\slshape\footnotesize \squelch{\myheadinglogos}]{\providecommand{\myheadinglogos}{\myheadinglogos}\slshape\footnotesize \squelch{\myheadinglogos}}%
    \cfoot[    \sffamily \footnotesize\thepage/\pageref{LastPage}]{    \sffamily \footnotesize\squelch{\thepage/\pageref{LastPage}}}%
    \ofoot[    \sffamily \slshape \footnotesize \squelch{ID 000000 (Platzhalter)}]{    \sffamily \slshape \footnotesize \squelch{ID 000000 (Platzhalter)}}%
    \ifoot[    \sffamily \slshape \footnotesize \squelch{\leftmark}]{    \sffamily \slshape \footnotesize \squelch{\rightmark}}%
    \setlength{\headheight}{1.33\baselineskip}%
    
    %% compatibility to old generic template:
    %\setheadsepline{0.5pt}[\vspace{-2pt}]%
    \addtolength{\topmargin}{-6pt}
    \addtolength{\headsep}{20pt}
    \addtolength{\textheight}{12pt}
    \addtolength{\footskip}{-12pt}
}
